
byogpm_decision_enable_precursors = {
    owned_planets_only = yes
    icon = decision_art_monument

    potential = {
        byogpm_base_enabled = yes
        NOT = { has_carrier_flag = byogpm_precursor_menu_open }
    }

    allow = {
        byogpm_precursor_base = yes
        custom_tooltip_fail = {
            text = byogpm_no_precursor_world
            NOT = { has_carrier_flag = gpm_is_precursor_world }
        }
    }

    effect = {
        custom_tooltip = byogpm_decision_enable_precursors.tooltip
        hidden_effect = { byogpm_gpm_precursor_planet_setup_no_modifiers = yes }
        set_carrier_flag = byogpm_precursor_menu_open
    }

    ai_weight = { weight = 0 }
}

byogpm_decision_disable_precursors = {
    owned_planets_only = yes
    icon = decision_art_monument

    potential = {
        byogpm_base_enabled = yes
        has_carrier_flag = byogpm_precursor_menu_open
    }

    effect = {
        custom_tooltip = byogpm_decision_disable_precursors.tooltip
        remove_carrier_flag = byogpm_precursor_menu_open
    }

    ai_weight = { weight = 0 }
}
