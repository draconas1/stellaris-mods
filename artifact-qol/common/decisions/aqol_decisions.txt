
aqol_decision_museum_exhibits_start = {
    owned_planets_only = yes
    sound = event_administrative_work
    icon = decision_museum

    potential = {
        owner = {
            is_gestalt = no
            OR = {
                is_megacorp = no
                is_worker_coop_empire = yes
            }
        }
        NOT = { has_designation = col_nanotech_world }
        NOT = { has_modifier = "aqol_planet_museum_exhibits" }
    }

    allow = {
        num_assigned_jobs = {
            job = bureaucrat
            value >= 100
        }
    }

    effect = {
        add_modifier = { modifier = "aqol_planet_museum_exhibits" days = -1 }
    }

    ai_weight = {
        weight = 0
    }
}
