trait_robot_digital_1 = {
	cost = 0
	category = robotic
	initial = no
	randomized = no
	allowed_archetypes = { ROBOT MACHINE }
	tags = { machine positive habitability housing special }
	opposites = {  }

	species_potential_add = {
		always = no
	}

	species_possible_remove = {
		always = no
	}

	modifier = {
		habitability_floor_add = 0.25
	}
	triggered_species_modifier = {
		potential = {
			exists = from
			from = {
				OR = {
					is_machine_empire = yes
					is_individual_machine = yes
				}
			}
		}
		pop_housing_usage_mult = -0.9
	}

	resources = {
		category = planet_pops
		# Energy Upkeep
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					has_tradition = tr_virtuality_1
				}
				is_sapient = yes
			}
			energy = 1	# All the maths is done in the script value
			mult = value:virtuality_num_colonies
		}
	}

	custom_tooltip_with_modifiers = trait_robot_digital_1_tt

	slave_cost = {
		trade = 750
	}
}
