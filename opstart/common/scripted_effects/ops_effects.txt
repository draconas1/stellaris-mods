ops_apply_modifier = {
    random_system_planet = {
        limit = {
            has_planet_flag = $FLAG$
            is_planet_class = $CLASS$
        }
        remove_planet_flag = $FLAG$
        add_modifier = { modifier = $MODIFIER$ }
    }
}

ops_fix_planet = {
    random_system_planet = {
        limit = { has_planet_flag = $FLAG$ }
        remove_planet_flag = $FLAG$
        clear_deposits = yes
        add_deposit = $DEPOSIT$
    }
}

ops_spawn_gpm_precursor_mod = {
    random_system_planet = {
        limit = {
            has_planet_flag = ops_outer_ring_small_2
            NOT = { is_planet_class = pc_gray_goo }
        }
        remove_planet_flag = ops_outer_ring_small_2
        set_planet_flag = gpm_is_precursor_world
        # roll precursor modifiers
        add_modifier = { days = -1 modifier = $MODIFIER$ multiplier = gpm_planet_size_multiplier_precursor }
        # spawn precursor features
        gpm_spawn_precursor_features = yes
        create_archaeological_site = gpm_precursor_generic_digsite
        set_planet_flag = gpm_already_rolled_for_prec_dig_site
    }
}
