calculate_energy_cost_for_some_pops = {
	base = 0
	modifier = {
		is_gestalt = no
		add = value:$RANKED_CALC$|CATEGORY|ruler|MULT|@baseEnergyPrice|
	}
	modifier = {
		is_gestalt = no
		add = value:$RANKED_CALC$|CATEGORY|specialist|MULT|@baseEnergyPrice|
	}
	modifier = {
		is_gestalt = no
		add = value:$RANKED_CALC$|CATEGORY|worker|MULT|@baseEnergyPrice|
	}
	modifier = {
		is_gestalt = no
		add = value:$RANKED_CALC$|CATEGORY|civilian|MULT|@baseEnergyPrice|
	}
	modifier = {
		is_gestalt = yes
		add = value:$RANKED_CALC$|CATEGORY|complex_drone|MULT|@baseEnergyPrice|
	}
	modifier = {
		is_gestalt = yes
		add = value:$RANKED_CALC$|CATEGORY|simple_drone|MULT|@baseEnergyPrice|
	}
	modifier = {
		is_gestalt = yes
		add = value:$RANKED_CALC$|CATEGORY|maintenance_drone|MULT|@baseEnergyPrice|
	}


	add = value:$OTHER_CALC$|MULT|@baseEnergyPrice|
	multiply = value:scripted_modifier_mult|MODIFIER|pop_resettlement_cost_mult|
	divide = @batchSize # we have counted pops, but they are moved in batches of 100 for the price
}

	base = 0
	modifier = {
		add = value:$RANKED_CALC$|CATEGORY|ruler|MULT|@baseUnityPrice|
		is_gestalt = no
		NOT = { has_valid_civic = civic_corvee_system }
	}
	modifier = {		
		add = value:$RANKED_CALC$|CATEGORY|specialist|MULT|@baseUnityPrice|
		is_gestalt = no
		NOT = { has_valid_civic = civic_corvee_system }
	}
	modifier = {
		add = value:$RANKED_CALC$|CATEGORY|worker|MULT|@baseUnityPrice|
		is_gestalt = no

	modifier = {
		add = value:$RANKED_CALC$|CATEGORY|civilian|MULT|@baseUnityPrice|
		is_gestalt = no

	}
	modifier = {
		add = value:$RANKED_CALC$|CATEGORY|complex_drone|MULT|@baseUnityPrice|
		is_gestalt = yes
		NOR = {
			has_valid_civic = civic_machine_ota_updates

		}

		NOR = {

			has_valid_civic = civic_hive_subsumed_will
		}
	}
	multiply = value:scripted_modifier_mult|MODIFIER|pop_resettlement_cost_mult|
	divide = @batchSize # we have counted pops, but they are moved in batches of 100 for the price
}
